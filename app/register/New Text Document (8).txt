// 1. دالة الفتح مع دعم "زر الرجوع" في الموبايل
function openMoFinal(packageName) { 
    var modal = document.getElementById("moModalFinal");
    if(modal) {
        modal.style.setProperty("display", "flex", "important"); 
        
        // اختيار الباقة تلقائياً إذا وجدت
        if (packageName) {
            var selectBox = document.getElementById("f_package");
            if(selectBox) selectBox.value = packageName;
        }

        // إضافة حالة للسجل لمنع الخروج من الموقع عند الضغط على "رجوع"
        history.pushState({ modalOpen: true }, "");
    }
}

// 2. دالة الإغلاق
function closeMoFinal() { 
    var modal = document.getElementById("moModalFinal");
    if(modal) {
        modal.style.setProperty("display", "none", "important");
        // الرجوع بالسجل خطوة للخلف إذا كانت النافذة مفتوحة
        if (history.state && history.state.modalOpen) {
            history.back();
        }
    }
}

// 3. معالجة زر الرجوع في الموبايل
window.onpopstate = function(event) {
    var modal = document.getElementById("moModalFinal");
    if (modal) {
        modal.style.setProperty("display", "none", "important");
    }
};

// 4. إغلاق النافذة عند الضغط في أي مكان فارغ (خارج الكارت)
window.onclick = function(event) {
    var modal = document.getElementById('moModalFinal');
    if (event.target == modal) {
        modal.style.setProperty("display", "none", "important");
        if (history.state && history.state.modalOpen) {
            history.back();
        }
    }
};
